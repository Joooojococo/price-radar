//@version=5
strategy("Gann Horizontal Line Strategy", overlay=true, initial_capital=10000, default_qty_type=strategy.percent_of_equity, default_qty_value=10)

// ==================== 輸入參數 ====================
// 江恩線參數
gannProximity = input.float(0.002, "江恩線接近閾值 (%)", minval=0.001, maxval=0.01, step=0.001) / 100
slBuffer = input.float(10, "止損緩衝 ($)", minval=5, maxval=50, step=5)

// 蠟燭圖形態參數
hammerBodyRatio = input.float(2.0, "錘頭線下影線/身體比例", minval=1.5, maxval=3.0, step=0.1)
engulfingBodyMin = input.float(0.1, "吞沒形態最小身體 (%)", minval=0.05, maxval=0.3, step=0.01) / 100

// 風險管理
riskPerTrade = input.float(1, "每筆交易風險 (%)", minval=0.5, maxval=5, step=0.5)

// ==================== 江恩水平線計算 ====================
// 計算江恩線：n² 如果是奇數直接用，如果是偶數則用 n²+1
// 根據當前價格範圍動態計算

var float[] gannLines = array.new_float(0)

// 每個 bar 重新計算江恩線（實際應用中可以優化為只在價格範圍變化時計算）
if barstate.islast
    array.clear(gannLines)
    
    minPrice = ta.lowest(low, 500)
    maxPrice = ta.highest(high, 500)
    
    // 找到合適的 n 範圍
    // 對於 BTC 價格約 70,000-100,000，n 約在 265-316 之間
    startN = math.floor(math.sqrt(minPrice * 0.9))
    endN = math.ceil(math.sqrt(maxPrice * 1.1))
    
    for i = startN to endN
        square = i * i
        gannLine = square % 2 == 1 ? square : square + 1
        
        if gannLine >= minPrice * 0.9 and gannLine <= maxPrice * 1.1
            array.push(gannLines, gannLine)

// 繪製江恩線
if barstate.islast
    for i = 0 to array.size(gannLines) - 1
        lineValue = array.get(gannLines, i)
        line.new(x1=bar_index - 500, y1=lineValue, x2=bar_index, y2=lineValue, 
                 color=color.new(color.blue, 70), width=1, style=line.style_dashed)

// ==================== 輔助函數 ====================
// 找到最近的江恩線
getNearestGannLine(price) =>
    nearest = 0.0
    minDist = 100000.0
    
    if array.size(gannLines) > 0
        for i = 0 to array.size(gannLines) - 1
            lineValue = array.get(gannLines, i)
            dist = math.abs(price - lineValue)
            if dist < minDist
                minDist := dist
                nearest := lineValue
    
    nearest

// 檢查價格是否接近江恩線
isNearGannLine(price) =>
    nearest = getNearestGannLine(price)
    if nearest > 0
        distancePct = math.abs(price - nearest) / nearest
        distancePct <= gannProximity
    else
        false

// 找到下一條江恩線
getNextGannLine(currentLine, direction) =>
    nextLine = 0.0
    if array.size(gannLines) > 0
        for i = 0 to array.size(gannLines) - 1
            lineValue = array.get(gannLines, i)
            if direction == "up" and lineValue > currentLine
                if nextLine == 0 or lineValue < nextLine
                    nextLine := lineValue
            else if direction == "down" and lineValue < currentLine
                if nextLine == 0 or lineValue > nextLine
                    nextLine := lineValue
    nextLine

// ==================== 蠟燭圖形態識別 ====================
// 錘頭線（長下影線，小身體在頂部）
isHammer() =>
    body = math.abs(close - open)
    upperShadow = high - math.max(open, close)
    lowerShadow = math.min(open, close) - low
    totalRange = high - low
    
    bodySize = body / totalRange
    upperShadowSize = upperShadow / totalRange
    lowerShadowRatio = lowerShadow / body
    bodyPosition = (math.min(open, close) - low) / totalRange
    
    totalRange > 0 and lowerShadowRatio >= hammerBodyRatio and upperShadowSize <= 0.1 and bodyPosition >= 0.7

// 看漲吞沒
isBullishEngulfing() =>
    prevBody = math.abs(close[1] - open[1])
    currBody = math.abs(close - open)
    
    isPrevBearish = close[1] < open[1]
    isCurrBullish = close > open
    engulfs = open < close[1] and close > open[1]
    
    isPrevBearish and isCurrBullish and engulfs and currBody > prevBody

// 看跌吞沒
isBearishEngulfing() =>
    prevBody = math.abs(close[1] - open[1])
    currBody = math.abs(close - open)
    
    isPrevBullish = close[1] > open[1]
    isCurrBearish = close < open
    engulfs = open > close[1] and close < open[1]
    
    isPrevBullish and isCurrBearish and engulfs and currBody > prevBody

// 黃昏之星（三根K線的頂部反轉形態）
isEveningStar() =>
    firstBullish = close[2] > open[2]
    
    secondBody = math.abs(close[1] - open[1])
    secondRange = high[1] - low[1]
    secondSmallBody = secondRange > 0 and secondBody / secondRange < 0.3
    
    gapsUp = low[1] > close[2]
    
    thirdBearish = close < open
    firstMidpoint = (open[2] + close[2]) / 2
    closesBelowMid = close < firstMidpoint
    
    firstBullish and secondSmallBody and gapsUp and thirdBearish and closesBelowMid

// ==================== 進場邏輯 ====================
var bool longCondition = false
var bool shortCondition = false
var float entryGannLine = 0.0

// 檢查是否接近江恩線
nearGann = isNearGannLine(close)
nearestGann = getNearestGannLine(close)
isAboveLine = close > nearestGann

// 做多條件：價格在江恩線上方，回測線，出現錘頭線或看漲吞沒
longCondition := nearGann and isAboveLine and (isHammer() or isBullishEngulfing())
entryGannLine := longCondition ? nearestGann : entryGannLine[1]

// 做空條件：價格在江恩線下方，反彈觸碰線，出現看跌吞沒或黃昏之星
shortCondition := nearGann and not isAboveLine and (isBearishEngulfing() or isEveningStar())
entryGannLine := shortCondition ? nearestGann : entryGannLine[1]

// ==================== 進場執行 ====================
// 做多
if longCondition and strategy.position_size == 0
    entryPrice = open
    stopLoss = low - slBuffer
    takeProfit1 = entryPrice + (entryPrice - stopLoss)
    
    strategy.entry("Long", strategy.long)
    strategy.exit("Long SL", "Long", stop=stopLoss)
    strategy.exit("Long TP1", "Long", limit=takeProfit1, qty_percent=50)
    
    // TP2 在下一條江恩線
    nextGannUp = getNextGannLine(nearestGann, "up")
    if nextGannUp > 0
        strategy.exit("Long TP2", "Long", limit=nextGannUp, qty_percent=50)

// 做空
if shortCondition and strategy.position_size == 0
    entryPrice = open
    stopLoss = high + slBuffer
    takeProfit1 = entryPrice - (stopLoss - entryPrice)
    
    strategy.entry("Short", strategy.short)
    strategy.exit("Short SL", "Short", stop=stopLoss)
    strategy.exit("Short TP1", "Short", limit=takeProfit1, qty_percent=50)
    
    // TP2 在下一條江恩線
    nextGannDown = getNextGannLine(nearestGann, "down")
    if nextGannDown > 0
        strategy.exit("Short TP2", "Short", limit=nextGannDown, qty_percent=50)

// ==================== Alert 設置 ====================
// 做多 Alert
alertcondition(longCondition, title="Gann Long Signal", message="Gann 做多信號：價格 {{close}} 接近江恩線 {{nearestGann}}")

// 做空 Alert
alertcondition(shortCondition, title="Gann Short Signal", message="Gann 做空信號：價格 {{close}} 接近江恩線 {{nearestGann}}")

// ==================== 視覺標記 ====================
// 標記進場點
plotshape(longCondition, title="Long Signal", location=location.belowbar, color=color.green, style=shape.triangleup, size=size.small)
plotshape(shortCondition, title="Short Signal", location=location.abovebar, color=color.red, style=shape.triangledown, size=size.small)

// 標記江恩線位置
plot(nearGann ? nearestGann : na, title="Nearest Gann Line", color=color.blue, linewidth=2, style=plot.style_circles)

// ==================== 資訊面板 ====================
var table infoTable = table.new(position.top_right, 2, 6, bgcolor=color.new(color.black, 80), border_width=1)

if barstate.islast
    table.cell(infoTable, 0, 0, "江恩線接近閾值", text_color=color.white)
    table.cell(infoTable, 1, 0, str.tostring(gannProximity * 100, "#.##%"), text_color=color.white)
    
    table.cell(infoTable, 0, 1, "止損緩衝", text_color=color.white)
    table.cell(infoTable, 1, 1, "$" + str.tostring(slBuffer), text_color=color.white)
    
    table.cell(infoTable, 0, 2, "最近江恩線", text_color=color.white)
    table.cell(infoTable, 1, 2, str.tostring(nearestGann, "#.##"), text_color=color.blue)
    
    table.cell(infoTable, 0, 3, "價格位置", text_color=color.white)
    table.cell(infoTable, 1, 3, isAboveLine ? "線上方" : "線下方", text_color=isAboveLine ? color.green : color.red)
    
    table.cell(infoTable, 0, 4, "錘頭線", text_color=color.white)
    table.cell(infoTable, 1, 4, isHammer() ? "是" : "否", text_color=isHammer() ? color.green : color.gray)
    
    table.cell(infoTable, 0, 5, "看漲吞沒", text_color=color.white)
    table.cell(infoTable, 1, 5, isBullishEngulfing() ? "是" : "否", text_color=isBullishEngulfing() ? color.green : color.gray)
